Current Endpoints
GET /menu/:id
	Returns menu information for restaurant with restaurant_id = id
	Output:
	{
		dish_name:
		{
			restaurant,
			calories,
			price,
			tax,
			options,
			category,
			breakfast,
			lunch,
			dinner,
			picture,
			in_stock
		}
	}

GET /restaurant/:id
	Returns restaurant information and menu information for restaurant with restaurant_id = id
	Output:
	{
		restaurant:
		{
			name,
			address,
			phone_number,
			opening,
			closing
		}
		menu:
		{
			dish_name:
			{
				calories,
				price,
				tax,
				options,
				category,
				breakfast,
				lunch,
				dinner,
				picture,
				in_stock
			}
		}
	}

GET /alexa/:id
	Returns information associated with the alexa with alexa_id = id
	Output:
	{
		alexa_id:
		{
			restaurant_id,
			table_num
		}
	}

GET /favorites/:id
	Returns information about a users favorite restaurants
	Output:
	{
		0:
		{
			restaurant_id,
			restaurant_name
		}
	}

POST /customers/login
	Logs a customer in and returns a signed JWT and customer information
	Parameters: In the BODY of the request send in the following where the values in {} are replaced
		username={user}&password={password}
	Output:
	{
		token:
		{
			token
		}
		user:
		{
			user_id,
			first_name,
			last_name,
			email
		}
	}

POST /staff/login
	Logs a customer in and returns a signed JWT and customer information
	Output:
	{
		token:
		{
			token
		}
		staff:
		{
			staff_id,
			restaurant_id,
			first_name,
			last_name,
			contact_num,
			email,
			position
		}
	}

POST /verify
	Verifies a JWT Token
	Output:
	{
		'message': 'Test Passed'
	}
